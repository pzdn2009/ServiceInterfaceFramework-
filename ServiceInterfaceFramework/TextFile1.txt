SELECT COLUMN_NAME, 
CASE WHEN  A.CHARACTER_MAXIMUM_LENGTH IS NULL THEN  data_type 
WHEN A.CHARACTER_MAXIMUM_LENGTH IS NOT NULL THEN 
data_type + '(' +convert(varchar(10), isnull( A.CHARACTER_MAXIMUM_LENGTH,0))  + ')'  END ,
CASE A.IS_NULLABLE WHEN 'NO' THEN 'N' 
 WHEN 'YES' THEN 'Y' END 
FROM INFORMATION_SCHEMA.columns AS A WHERE TABLE_NAME = 'data_order'
